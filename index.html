// ====================== ABRIR MATÉRIA ====================== function abrirMateria(index) { materiaAtual = { index: index, // Não usamos spread aqui para evitar cópia desnecessária // Vamos acessar materias[index] diretamente quando precisar alterar }; // Atualizamos as referências materiaAtual.nome = materias[index].nome; materiaAtual.anotacoes = materias[index].anotacoes; // mesma referência! document.getElementById('telaInicial').classList.add('hidden'); document.getElementById('areaAnotacoes').classList.remove('hidden'); document.getElementById('headerMateria').classList.remove('hidden'); document.getElementById('tituloMateria').textContent = materiaAtual.nome; atualizarContador(); renderizarAnotacoes(); } // ====================== SALVAR ANOTAÇÃO ====================== function salvarAnotacao() { if (!materiaAtual) return; const textarea = document.getElementById('novaAnotacao'); const texto = textarea.value.trim(); if (texto === '') { alert("Escreva algo antes de salvar!"); return; } const agora = new Date(); const dataFormatada = agora.toLocaleDateString('pt-BR', { day: '2-digit', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit' }); // Adicionamos diretamente no array original materias[materiaAtual.index].anotacoes.unshift({ texto: texto, data: dataFormatada }); salvarDados(); // Como anotacoes é a mesma referência, já está atualizado textarea.value = ''; renderizarAnotacoes(); atualizarContador(); } // ====================== EXCLUIR ANOTAÇÃO ====================== function excluirAnotacao(indice) { if (!confirm("Excluir esta anotação?")) return; materias[materiaAtual.index].anotacoes.splice(indice, 1); salvarDados(); // Mesma referência → já atualiza materiaAtual também renderizarAnotacoes(); atualizarContador(); } // ====================== RENDERIZAR ANOTAÇÕES ====================== function renderizarAnotacoes() { const container = document.getElementById('listaAnotacoes'); container.innerHTML = ''; if (materiaAtual.anotacoes.length === 0) { container.innerHTML = `

Ainda não há anotações...

Escreva acima e clique em Salvar

`; return; } materiaAtual.anotacoes.forEach((anotacao, i) => { const div = document.createElement('div'); div.className = `anotacao bg-zinc-900 border border-zinc-800 rounded-3xl p-8 group hover:border-violet-500/30`; div.innerHTML = `
${anotacao.data}

${anotacao.texto}

`; // Adicionamos o listener de exclusão aqui (melhor prática) div.querySelector('button').addEventListener('click', (e) => { e.stopPropagation(); // impede que o clique no card seja disparado excluirAnotacao(i); }); container.appendChild(div); }); }
