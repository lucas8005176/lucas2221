let materias = []; let materiaAtual = null; // ====================== CARREGAR E SALVAR ====================== function carregarDados() { const salvos = localStorage.getItem('materias'); materias = salvos ? JSON.parse(salvos) : []; renderizarListaMaterias(); } function salvarDados() { localStorage.setItem('materias', JSON.stringify(materias)); } // ====================== RENDERIZAR LISTA DE MATÉRIAS (SIDEBAR) ====================== function renderizarListaMaterias() { const container = document.getElementById('listaMaterias'); container.innerHTML = ''; if (materias.length === 0) { container.innerHTML = `

Nenhuma matéria ainda

Crie uma acima para começar

`; return; } materias.forEach((materia, index) => { const isActive = materiaAtual && materiaAtual.index === index; const div = document.createElement('div'); div.className = `materia-card mx-3 my-1 px-6 py-5 rounded-3xl cursor-pointer flex items-center gap-4 transition-all ${isActive ? 'bg-violet-900/30 border border-violet-500' : 'hover:bg-zinc-800'}`; div.innerHTML = `

${materia.nome}

${materia.anotacoes.length} anotação${materia.anotacoes.length !== 1 ? 's' : ''}

`; div.onclick = () => abrirMateria(index); container.appendChild(div); }); } // ====================== ADICIONAR MATÉRIA ====================== function adicionarMateria() { const input = document.getElementById('novaMateria'); const nome = input.value.trim(); if (nome === '') { alert("Digite o nome da matéria!"); input.focus(); return; } // Verifica se já existe if (materias.some(m => m.nome.toLowerCase() === nome.toLowerCase())) { alert("Essa matéria já existe!"); return; } // Adiciona no início da lista materias.unshift({ nome: nome, anotacoes: [] }); salvarDados(); renderizarListaMaterias(); // Limpa o campo input.value = ''; input.focus(); } // ====================== ABRIR MATÉRIA ====================== function abrirMateria(index) { materiaAtual = { index: index, ...materias[index] }; // Esconde tela inicial e mostra área de anotações document.getElementById('telaInicial').classList.add('hidden'); document.getElementById('areaAnotacoes').classList.remove('hidden'); document.getElementById('headerMateria').classList.remove('hidden'); document.getElementById('tituloMateria').textContent = materiaAtual.nome; atualizarContador(); renderizarAnotacoes(); } function voltarParaMaterias() { materiaAtual = null; document.getElementById('telaInicial').classList.remove('hidden'); document.getElementById('areaAnotacoes').classList.add('hidden'); document.getElementById('headerMateria').classList.add('hidden'); renderizarListaMaterias(); } function atualizarContador() { const contador = document.getElementById('contadorAnotacoes'); const qtd = materiaAtual.anotacoes.length; contador.textContent = `${qtd} anotação${qtd !== 1 ? 's' : ''}`; } // ====================== SALVAR ANOTAÇÃO ====================== function salvarAnotacao() { if (!materiaAtual) return; const textarea = document.getElementById('novaAnotacao'); const texto = textarea.value.trim(); if (texto === '') { alert("Escreva algo antes de salvar!"); return; } const agora = new Date(); const dataFormatada = agora.toLocaleDateString('pt-BR', { day: '2-digit', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit' }); materias[materiaAtual.index].anotacoes.unshift({ texto: texto, data: dataFormatada }); salvarDados(); materiaAtual.anotacoes = materias[materiaAtual.index].anotacoes; textarea.value = ''; renderizarAnotacoes(); atualizarContador(); } // ====================== RENDERIZAR ANOTAÇÕES ====================== function renderizarAnotacoes() { const container = document.getElementById('listaAnotacoes'); container.innerHTML = ''; if (materiaAtual.anotacoes.length === 0) { container.innerHTML = `

Ainda não há anotações...

Escreva acima e clique em Salvar

`; return; } materiaAtual.anotacoes.forEach((anotacao, i) => { const div = document.createElement('div'); div.className = `anotacao bg-zinc-900 border border-zinc-800 rounded-3xl p-8 group hover:border-violet-500/30`; div.innerHTML = `
${anotacao.data}

${anotacao.texto}

`; container.appendChild(div); }); } function excluirAnotacao(indice) { if (!confirm("Excluir esta anotação?")) return; materias[materiaAtual.index].anotacoes.splice(indice, 1); salvarDados(); materiaAtual.anotacoes = materias[materiaAtual.index].anotacoes; renderizarAnotacoes(); atualizarContador(); } // ====================== LIMPAR TUDO ====================== function limparTudo() { if (confirm("⚠️ APAGAR TODAS as matérias e anotações?\n\nEssa ação não pode ser desfeita!")) { localStorage.removeItem('materias'); materias = []; materiaAtual = null; voltarParaMaterias(); renderizarListaMaterias(); } } // ====================== INICIALIZAR ====================== window.onload = () => { carregarDados(); // Permite adicionar com Enter no input const inputMateria = document.getElementById('novaMateria'); inputMateria.addEventListener('keypress', (e) => { if (e.key === 'Enter') adicionarMateria(); }); };
