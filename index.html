let materiaAtual = null; let materias = []; // Carregar dados function carregarDados() { materias = JSON.parse(localStorage.getItem('materias')) || []; renderizarMaterias(); } function salvarDados() { localStorage.setItem('materias', JSON.stringify(materias)); } // Renderizar sidebar function renderizarMaterias() { const container = document.getElementById('listaMaterias'); container.innerHTML = ''; if (materias.length === 0) { container.innerHTML = `

Nenhuma matéria ainda

`; return; } materias.forEach((materia, index) => { const div = document.createElement('div'); div.className = `materia-card mx-3 mb-2 px-5 py-4 rounded-3xl cursor-pointer flex items-center gap-4 ${materiaAtual && materiaAtual.index === index ? 'bg-violet-900/30 border border-violet-500' : 'hover:bg-zinc-800'}`; div.innerHTML = `

${materia.nome}

${materia.anotacoes.length} anotação${materia.anotacoes.length !== 1 ? 's' : ''}

`; div.onclick = () => abrirMateria(index); container.appendChild(div); }); } // Adicionar matéria function adicionarMateria() { const input = document.getElementById('novaMateria'); const nome = input.value.trim(); if (!nome) return; if (materias.some(m => m.nome.toLowerCase() === nome.toLowerCase())) { alert("Essa matéria já existe!"); return; } materias.unshift({ nome: nome, anotacoes: [] }); salvarDados(); renderizarMaterias(); input.value = ''; } // Abrir matéria function abrirMateria(index) { materiaAtual = { index, ...materias[index] }; document.getElementById('telaInicial').classList.add('hidden'); document.getElementById('areaAnotacoes').classList.remove('hidden'); document.getElementById('headerMateria').classList.remove('hidden'); document.getElementById('tituloMateria').textContent = materiaAtual.nome; document.getElementById('contadorAnotacoes').textContent = `${materiaAtual.anotacoes.length} anotação${materiaAtual.anotacoes.length !== 1 ? 's' : ''}`; renderizarAnotacoes(); } function voltarParaMaterias() { materiaAtual = null; document.getElementById('areaAnotacoes').classList.add('hidden'); document.getElementById('headerMateria').classList.add('hidden'); document.getElementById('telaInicial').classList.remove('hidden'); renderizarMaterias(); } // Salvar anotação function salvarAnotacao() { if (!materiaAtual) return; const texto = document.getElementById('novaAnotacao').value.trim(); if (!texto) return alert("Escreva algo antes de salvar!"); const data = new Date().toLocaleDateString('pt-BR', { day: '2-digit', month: 'short', year: 'numeric', hour: '2-digit', minute: '2-digit' }); materias[materiaAtual.index].anotacoes.unshift({ texto: texto, data: data }); salvarDados(); materiaAtual.anotacoes = materias[materiaAtual.index].anotacoes; document.getElementById('novaAnotacao').value = ''; renderizarAnotacoes(); document.getElementById('contadorAnotacoes').textContent = `${materiaAtual.anotacoes.length} anotação${materiaAtual.anotacoes.length !== 1 ? 's' : ''}`; } // Renderizar anotações function renderizarAnotacoes() { const container = document.getElementById('listaAnotacoes'); container.innerHTML = ''; if (materiaAtual.anotacoes.length === 0) { container.innerHTML = `

Nenhuma anotação ainda...

Escreva algo acima e salve!

`; return; } materiaAtual.anotacoes.forEach((anotacao, i) => { const div = document.createElement('div'); div.className = 'anotacao bg-zinc-900 border border-zinc-800 rounded-3xl p-7 group'; div.innerHTML = `
${anotacao.data}

${anotacao.texto}

`; container.appendChild(div); }); } function excluirAnotacao(indice) { if (!confirm("Excluir esta anotação permanentemente?")) return; materias[materiaAtual.index].anotacoes.splice(indice, 1); salvarDados(); materiaAtual.anotacoes = materias[materiaAtual.index].anotacoes; renderizarAnotacoes(); document.getElementById('contadorAnotacoes').textContent = `${materiaAtual.anotacoes.length} anotação${materiaAtual.anotacoes.length !== 1 ? 's' : ''}`; } function limparTudo() { if (confirm("⚠️ Isso vai apagar TODAS as matérias e anotações. Tem certeza?")) { localStorage.removeItem('materias'); materias = []; voltarParaMaterias(); renderizarMaterias(); } } // Inicializar window.onload = () => { carregarDados(); };
